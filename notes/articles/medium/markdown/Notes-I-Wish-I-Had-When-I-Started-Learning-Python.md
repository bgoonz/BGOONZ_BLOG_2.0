    sentence = 'Mary, Mary, quite contrary, how does your garden grow?'

    consonants = [i for i in sentence if is_consonant(i)]

    print(consonants)
    # Prints ['M', 'r', 'y', 'M', 'r', 'y', 'q', 't', 'c',
